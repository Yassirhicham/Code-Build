#define BLYNK_PRINT Serial
#include <WiFi.h>

char ssid[] = "wifi_iris";

void setup() {
    Serial.begin(115200);
    Blynk.begin(auth, ssid, pass);
}

void loop() {
    Blynk.run();
}
